Build out the controller to for movies.
app/controllers/api/movies_controller.rb
class Api::MoviesController < ApplicationController
before_action :set_movie, except: [:index, :create]
def index
movies = Movie.all
render json: movies
end
def show
render json: @movie
end
def create
movie = Movie.new(movie_params)
if movie.save
render json: movie
else
render json: {errors: movie.errors}, status: :unprocessable_entity
end
end
def update
if @movie.update(movie_params)
render json: @movie
else
render json: {errors: @movie.errors}, status: :unprocessable_entity
end
end
def destroy
@movie.destroy
render json: "deleted"
end
private
def movie_params
params.require(:movie).permit(:title, :year, :description, :in_theaters)
end
def set_movie
@movie = Movie.find(params[:id])
end
end
webpack/components/Movies.js
import React, { Component } from 'react';
import { Link } from 'react-router';
class Movies extends React.Component {
constructor(props) {
super(props);
this.state = { movies: [] };
}
componentWillMount() {
$.ajax({
url: '/api/movies',
type: 'GET',
dataType: 'JSON'
}).done(movies => {
this.setState({ movies });
}).fail(data => {
// TODO - handle better
console.log(data);
});
}
displayMovies() {
let movies = this.state.movies.map( movie => {
return(<li key={movie.id}>
<Link to={`/movies/${movie.id}`}>
{movie.title}
</Link>
</li>
);
});
return movies;
}
render() {
return(
<div>
<h1>All Movies</h1>
<ul>
{ this.displayMovies() }
</ul>
</div>
)
}
}
export default Movies;

import React, { Component } from 'react';
class Movie extends Component {
constructor(props) {
super(props);
this.state = { movie: {} } };
}
componentWillMount() {
$.ajax({
url: `/api/movies/${this.props.params.id}`,
type: 'GET',
dataType: 'JSON'
}).done( movie => {
this.setState({ movie });
}).fail( data => {
// TODO: handle this better
console.log(data);
})
}
render() {

}
}
